Dangbei v2/chat Sign Test

Open this file on ai.dangbei.com domain via DevTools console, or use the console intercept script.

Instructions:

  1. Open ai.dangbei.com in Chrome
  2. Open DevTools (F12) → Console
  3. Paste the script below and press Enter
  4. Send any chat message on the page
  5. Copy the JSON output from console

Console Script:

// Intercept v2/chat and log ALL request details
(function() {
    const origFetch = window.fetch;
    const origXHR = XMLHttpRequest.prototype.open;
    
    // Intercept fetch
    window.fetch = async function(input, init) {
        const req = input instanceof Request ? input : new Request(input, init);
        const url = req.url || (typeof input === 'string' ? input : '');
        
        if (url.includes('/v2/chat')) {
            const headers = {};
            for (const [k, v] of req.headers.entries()) headers[k] = v;
            const body = await req.clone().text();
            
            console.log('%c=== INTERCEPTED v2/chat ===', 'color:green;font-weight:bold');
            console.log('URL:', url);
            console.log('Method:', req.method);
            console.log('Headers:', headers);
            console.log('Body:', body);
            
            // Output as copyable JSON
            const data = { url, method: req.method, headers, body };
            console.log('%cCOPY THIS:', 'color:red;font-weight:bold');
            console.log(JSON.stringify(data, null, 2));
            
            // Also copy to clipboard
            try {
                await navigator.clipboard.writeText(JSON.stringify(data, null, 2));
                console.log('(Copied to clipboard!)');
            } catch(e) {}
        }
        return origFetch.apply(this, arguments);
    };
    
    console.log('%cInterceptor ready. Send a chat message now.', 'color:blue;font-weight:bold');
})();
